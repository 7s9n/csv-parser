269  CSV_INLINE void CSVReader::read_csv(const size_t& bytes) {
270  const size_t BUFFER_UPPER_LIMIT = std::min(bytes, (size_t)1000000);
271  std::unique_ptr<char[]> buffer(new char[BUFFER_UPPER_LIMIT]);
272  auto * HEDLEY_RESTRICT line_buffer = buffer.get();
273  line_buffer[0] = '\0';
274 
275  std::thread worker(&CSVReader::read_csv_worker, this);
276 
277  for (size_t processed = 0; processed < bytes; ) {
278  char * HEDLEY_RESTRICT result = std::fgets(line_buffer, internals::PAGE_SIZE, this->infile);
279  if (result == NULL) break;
280  line_buffer += std::strlen(line_buffer);
281  size_t current_strlen = line_buffer - buffer.get();
282 
283  if (current_strlen >= 0.9 * BUFFER_UPPER_LIMIT) {
284  processed += (line_buffer - buffer.get());
285  std::unique_lock<std::mutex> lock{ this->feed_state->feed_lock };
286 
287  this->feed_state->feed_buffer.push_back(std::make_pair<>(std::move(buffer), current_strlen));
288 
289  buffer = std::unique_ptr<char[]>(new char[BUFFER_UPPER_LIMIT]); // New pointer
290  line_buffer = buffer.get();
291  line_buffer[0] = '\0';
292 
293  this->feed_state->feed_cond.notify_one();
294  }
295  }
296 
297  // Feed remaining bits
298  std::unique_lock<std::mutex> lock{ this->feed_state->feed_lock };
299  this->feed_state->feed_buffer.push_back(std::make_pair<>(std::move(buffer), line_buffer - buffer.get()));
300  this->feed_state->feed_buffer.push_back(std::make_pair<>(nullptr, 0)); // Termination signal
301  this->feed_state->feed_cond.notify_one();
302  lock.unlock();
303  worker.join();
304 
305  if (std::feof(this->infile)) {
306  this->end_feed();
307  this->close();
308  }
309  }
