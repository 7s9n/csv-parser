194  void parse_field(string_view in, size_t& i) noexcept {
195  using internals::ParseFlags;
196 
197  // Trim off leading whitespace
198  while (i < in.size() && ws_flag(in[i])) i++;
199 
200  if (this->field_start < 0) {
201  this->field_start = (int)(i - current_row_start());
202  }
203 
204  // Optimization: Since NOT_SPECIAL characters tend to occur in contiguous
205  // sequences, use the loop below to avoid having to go through the outer
206  // switch statement as much as possible
207  while (i < in.size() && compound_parse_flag(in[i]) == ParseFlags::NOT_SPECIAL) i++;
208 
209  this->field_length = i - (this->field_start + current_row_start());
210 
211  // Trim off trailing whitespace, this->field_length constraint matters
212  // when field is entirely whitespace
213  for (size_t j = i - 1; ws_flag(in[j]) && this->field_length > 0; j--) this->field_length--;
214  }
